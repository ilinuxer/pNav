示例
 var items = [
        {label: 'pNavi'}, 
        {label: '点击该面包屑', cb: function(label) {
              alert(item);
            }
        }, 
        {label: '面包屑导航条'}, 
        {label: '当前位置'}
    ];
var nav = new pNav({items: items});
文档说明:
var nav = new pNav({
    items: [
        {   label: 面包屑显示内容,
            cb: function(label){
                // 调用func
            }
        }
      ],  // 各级内容
    maxLevelCount: 4 // 最多显示级数
});
$el // 获取元素
addClass(), removeClass(),hove(), hide() ... // 参考jQuery
add(item, index, enable) // 添加选项: 内容, 位置, 是否覆盖
back()  // 返回上一级
beforeChange(cb)  // 导航内容改变前触发func
change(cb)  // 导航内容改变触发func
destroy() // 销毁
enableItem(index, enable) // 销毁某一级导航： 索引, 是否消除， false隐藏
forward() // 返回导航上一次位置
has(arg) // 检测是否存在对应的导航, 参数可为索引或字符串
item(index) // 根据索引返回一级数据
items() // 返回所有数据
lastDepth() // 上一次的深度
max() // 返回最多显示级数
navi(arg) // 改变导航内容: items值
next() // 下一步 与forward相反
update(label, newLabel) // 跟新其中一个导航
reset() // 重置
size() // 返回导航栏长度
